683 void Vortex::handleNumber(char c)
684 {
685  if (!isdigit(c) || !m_lastCommand) {
686  return;
687  }
688  int number = c - '0';
689  char newc = 0;
690  // read the digits into the repeatAmount
691  while (1) {
692  newc = getchar();
693  if (!isdigit(newc)) {
694  // stop once we reach a non digit
695  break;
696  }
697  // accumulate the digits into the repeat amount
698  number = (number * 10) + (newc - '0');
699  }
700  // shove the last non-digit back into the stream
701  ungetc(newc, stdin);
702  DEBUG_LOGF("Repeating last command (%c) x%u times", m_lastCommand, number);
703  m_commandLog += to_string(number);
704  // check to see if we are repeating a 'rapid click' which is a special case
705  // because the rapid click command itself is composed or 'r' and a repeat count
706  // to designate how many rapid clicks to deliver
707  switch (m_lastCommand) {
708  case 'r':
709  // perform a rapid click number times
710  rapidClick(number);
711  break;
712  case 'b':
713  // select button index to target for click events
714  selectButton(number);
715  break;
716  default: // any other command
717  if (number > 0) {
718  // offset repeat amount by exactly 1 because it's already done
719  number--;
720  }
721  // repeat the last command that many times
722  while (number > 0) {
723  doCommand(m_lastCommand);
724  number--;
725  }
726  break;
727  }
728 }

730 // injects a command into the engine, the engine will parse one command
731 // per tick so multiple commands will be queued up
732 void Vortex::doCommand(char c)
733 {
734  if (!isprint(c)) {
735  return;
736  }
737 
738  m_commandLog += c;
739 
740  switch (c) {
741  case 'c':
742  if (m_lastCommand != c) {
743  DEBUG_LOG("Injecting short click");
744  }
745  shortClick();
746  break;
747  case 'l':
748  if (m_lastCommand != c) {
749  DEBUG_LOG("Injecting long click");
750  }
751  longClick();
752  break;
753  case 'm':
754  if (m_lastCommand != c) {
755  DEBUG_LOG("Injecting menu enter click");
756  }
757  menuEnterClick();
758  break;
759  case 'a':
760  if (m_lastCommand != c) {
761  DEBUG_LOG("Injecting adv menu enter click");
762  }
763  advMenuEnterClick();
764  break;
765  case 'd':
766  if (m_lastCommand != c) {
767  DEBUG_LOG("Injecting adv menu enter click");
768  }
769  deleteColClick();
770  break;
771  case 's':
772  if (m_lastCommand != c) {
773  DEBUG_LOG("Injecting sleep click");
774  }
775  sleepClick();
776  break;
777  case 'f':
778  if (m_lastCommand != c) {
779  DEBUG_LOG("Injecting force sleep click");
780  }
781  forceSleepClick();
782  break;
783  case 'q':
784  //case '\n':
785  if (m_lastCommand != c) {
786  DEBUG_LOG("Injecting quit click\n");
787  }
788  quitClick();
789  break;
790  case 't':
791  if (isButtonPressed()) {
792  if (m_lastCommand != c) {
793  DEBUG_LOG("Injecting release");
794  }
795  releaseButton();
796  } else {
797  if (m_lastCommand != c) {
798  DEBUG_LOG("Injecting press");
799  }
800  pressButton();
801  }
802  break;
803  case 'r':
804  // do nothing for the initial 'r', handleNumber() will handle the numeric
805  // repeat count that follows this letter and issue the rapidClick(amt) call
806  break;
807  case 'b':
808  // button selector, same as rapid click, it's a little weird because the number
809  // it not a repeat amount but actually the button index to select -- but none
810  // the less it works the same as rapid click so it will be handled inside handleNumber()
811  break;
812  case 'w':
813  if (m_lastCommand != c) {
814  DEBUG_LOG("Injecting wait");
815  }
816  sendWait();
817  break;
818  default:
819  handleNumber(c);
820  // return instead of break because this isn't a command
821  return;
822  }
823  m_lastCommand = c;
824 }

848 bool Vortex::tick()
849 {
850  if (!m_initialized) {
851  // do not cleanup prematurely here
852  return false;
853  }
854  // use ioctl to determine how many characters are on stdin so that
855  // we don't call getchar() too many times and accidentally block
856  uint32_t numInputs = getNumInputs();
857  if (m_lockstepEnabled && !numInputs) {
858  // don't tick till we have input
859  return true;
860  }
861  // iterate the number of inputs on stdin and parse each letter
862  // into a command for the engine, this will inject all of the commands
863  // that are available into the engine but that doesn't necessarily
864  // mean that the engine will do anything with them right away
865  for (uint32_t i = 0; i < numInputs; ++i) {
866  doCommand(getchar());
867  }
868  // tick the vortex engine forward
869  VortexEngine::tick();
870  return true;
871 }
