39 bool EditorConnection::receiveMessage(const char *message)
40 {
41  size_t len = strlen(message);
42  uint8_t byte = 0;
43  // wait for the editor to ack the idle
44  if (m_receiveBuffer.size() < len) {
45  return false;
46  }
47  if (memcmp(m_receiveBuffer.frontUnserializer(), message, len) != 0) {
48  return false;
49  }
50  for (size_t i = 0; i < len; ++i) {
51  m_receiveBuffer.unserialize(&byte);
52  }
53  // if everything was read out, reset
54  if (m_receiveBuffer.unserializerAtEnd()) {
55  m_receiveBuffer.clear();
56  }
57  return true;
58 }

90  break;

94  SerialComs::write(EDITOR_VERB_GREETING);

99  handleCommand();

105  break;

107  // recive the send modes ack from the editor

109  m_state = STATE_PULL_MODES_DONE;
110  }
111  break;
112  case STATE_PULL_MODES_DONE:
113  m_receiveBuffer.clear();
114  // send our acknowledgement that the modes were sent
115  SerialComs::write(EDITOR_VERB_PULL_MODES_ACK);

121  m_receiveBuffer.clear();

222 bool EditorConnection::receiveModes()
223 {
224  // need at least the buffer size first
225  uint32_t size = 0;
226  if (m_receiveBuffer.size() < sizeof(size)) {
227  // wait, not enough data available yet
228  return false;
229  }
230  // grab the size out of the start
231  m_receiveBuffer.resetUnserializer();
232  size = m_receiveBuffer.peek32();
233  if (m_receiveBuffer.size() < (size + sizeof(size))) {
234  // don't unserialize yet, not ready
235  return false;
236  }
237  // okay unserialize now, first unserialize the size
238  m_receiveBuffer.unserialize(&size);
239  // create a new ByteStream that will hold the full buffer of data
240  ByteStream buf(m_receiveBuffer.rawSize());
241  // then copy everything from the receive buffer into the rawdata
242  // which is going to overwrite the crc/size/flags of the ByteStream
243  memcpy(buf.rawData(), m_receiveBuffer.data() + sizeof(size),
244  m_receiveBuffer.size() - sizeof(size));
245  // clear the receive buffer
246  m_receiveBuffer.clear();
247  Modes::loadFromBuffer(buf);
248  Modes::saveStorage();
249  return true;
250 }
