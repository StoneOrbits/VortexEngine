154 void IRReceiver::handleIRTiming(uint32_t diff)
155 {
156  // if the diff is too long or too short then it's not useful
157  if ((diff > IR_HEADER_MARK_MAX && m_recvState < READING_DATA_MARK) || diff < IR_TIMING_MIN) {
158  DEBUG_LOGF("bad delay: %u, resetting...", diff);
159  resetIRState();
160  return;
161  }
162  switch (m_recvState) {
163  case WAITING_HEADER_MARK: // initial state
164  if (diff >= IR_HEADER_MARK_MIN && diff <= IR_HEADER_MARK_MAX) {
165  m_recvState = WAITING_HEADER_SPACE;
166  } else {
167  DEBUG_LOGF("Bad header mark %u, resetting...", diff);
168  resetIRState();
169  }
170  break;
171  case WAITING_HEADER_SPACE:
172  if (diff >= IR_HEADER_SPACE_MIN && diff <= IR_HEADER_SPACE_MAX) {
173  m_recvState = READING_DATA_MARK;
174  } else {
175  DEBUG_LOGF("Bad header space %u, resetting...", diff);
176  resetIRState();
177  }

179  case READING_DATA_MARK:
