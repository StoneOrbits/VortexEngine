112 void ColorSelect::onLongClick()
113 {
114  // if we're on 'exit' and we're on any menu past the slot selection
115  if (m_curSelection == 4 && m_state > STATE_PICK_SLOT) {
116  // move back to the previous selection
117  m_state = (ColorSelectState)(m_state - 1);
118  // if we're back to the slot selection then set our position back to the
119  // slot we selected, otherwise it's more visually appealing to just return
120  // to the first selection on the previous menu
121  m_curSelection = (m_state == STATE_PICK_SLOT) ? m_targetSlot : 0;
122  return;
123  }
124  // reuse these variables lots
125  uint8_t numColors = m_colorset.numColors();
126  uint32_t holdDur = g_pButton->holdDuration();
127  switch (m_state) {
128  case STATE_INIT:
129  // nothing
130  return;
131  case STATE_PICK_SLOT:
132  // if the exit is selected then save and exit, this depends on the number
133  // of colors in the colorset. If the colorset is full then the exit is
134  // just the number of colors (8) but if it's not full then the exit is the
135  // number of colors + 1. Example: with 4 cols, cols are on 0, 1, 2, 3,
136  // add-color is 4, and exit is 5
137  if (m_curSelection == numColors + (numColors < MAX_COLOR_SLOTS)) {
138  Mode *cur = Modes::curMode();
139  cur->setColorsetMap(m_targetLeds, m_colorset);
140  cur->init();
141  leaveMenu(true);
142  return;
143  }
144  // handle if user releases during the delete option
145  if (m_curSelection < numColors &&
146  holdDur >= DELETE_THRESHOLD_TICKS &&
147  (holdDur % (DELETE_CYCLE_TICKS * 2)) > (DELETE_CYCLE_TICKS)) {
148  // delete current slot
149  m_colorset.removeColor(m_curSelection);
150  if (m_curSelection > numColors) {
151  m_curSelection--;
152  }
153  return;
154  }
155  // otherwise store the target slot continue onto the hue selection
156  m_targetSlot = m_curSelection;
157  break;
158  case STATE_PICK_HUE1:
159  m_targetHue1 = m_curSelection;
160  m_newColor.hue = m_targetHue1 * (255 / 4);
161  break;
162  case STATE_PICK_HUE2:
163  m_newColor.hue = (m_targetHue1 * (255 / 4)) + m_curSelection * (255 / 16);
164  break;
165  case STATE_PICK_SAT:
166  m_newColor.sat = sats[m_curSelection];
167  break;
168  case STATE_PICK_VAL:
169  // no m_targetVal because you can't go back after this
170  m_newColor.val = vals[m_curSelection];
171  // specifically using hsv to rgb rainbow to generate the color
172  m_colorset.set(m_targetSlot, m_newColor);
173  m_curSelection = m_targetSlot;
174  m_state = STATE_PICK_SLOT;
175  return;
176  }
177  m_state = (ColorSelectState)(m_state + 1);
178  m_curSelection = 0;
179 }
