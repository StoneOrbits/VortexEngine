84 {
85  // Sometimes the pattern needs to cycle multiple states in a single frame so
86  // instead of using a loop or recursion I have just used a simple goto
87 replay:
88 
89  // its kinda evolving as i go
90  switch (m_state) {
91  case STATE_DISABLED:
92  return;
93  case STATE_BLINK_ON:
94  if (m_onDuration > 0) {
95  onBlinkOn();
96  --m_groupCounter;
97  nextState(m_onDuration);
98  return;
99  }
100  m_state = STATE_BLINK_OFF;
101  case STATE_BLINK_OFF:
102  // the whole 'should blink off' situation is tricky because we might need
103  // to go back to blinking on if our colorset isn't at the end yet
104  if (m_groupCounter > 0 || (!m_gapDuration && !m_dashDuration)) {
105  if (m_offDuration > 0) {
106  onBlinkOff();
107  nextState(m_offDuration);
108  return;
109  }
110  if (m_groupCounter > 0 && m_onDuration > 0) {
111  m_state = STATE_BLINK_ON;
112  goto replay;
113  }
114  }
115  m_state = STATE_BEGIN_GAP;
116  case STATE_BEGIN_GAP:
117  m_groupCounter = m_groupSize ? m_groupSize : (m_colorset.numColors() - (m_dashDuration != 0));
118  if (m_gapDuration > 0) {
119  beginGap();
120  nextState(m_gapDuration);
121  return;
122  }
123  m_state = STATE_BEGIN_DASH;
124  case STATE_BEGIN_DASH:
125  if (m_dashDuration > 0) {
126  beginDash();
127  nextState(m_dashDuration);
128  return;
129  }
130  m_state = STATE_BEGIN_GAP2;
131  case STATE_BEGIN_GAP2:
132  if (m_dashDuration > 0 && m_gapDuration > 0) {
133  beginGap();
134  nextState(m_gapDuration);
135  return;
136  }
137  m_state = STATE_BLINK_ON;
138  goto replay;
139  default:
140  break;
141  }
142 
143  if (m_blinkTimer.alarm() == -1) {
144  // no alarm triggered just stay in current state, return and don't transition states
145  PRINT_STATE(m_state);
146  return;
147  }
148 
149  // this just transitions the state into the next state, with some edge conditions for
150  // transitioning to different states under certain circumstances. Honestly this is
151  // a nightmare to read now and idk how to fix it
152  if (m_state == STATE_IN_GAP2 || (m_state == STATE_OFF && m_groupCounter > 0)) {
153  // this is an edge condition for when in the second gap or off in the non-last off blink
154  // then the state actually needs to jump backwards rather than iterate
155  m_state = m_onDuration ? STATE_BLINK_ON : (m_dashDuration ? STATE_BEGIN_DASH : STATE_BEGIN_GAP);
156  } else if (m_state == STATE_OFF && (!m_groupCounter || m_colorset.numColors() == 1)) {
157  // this is an edge condition when the state is off but this is the last off blink in the
158  // group or there's literally only one color in the group then if there is more blinks
159  // left in the group we need to cycle back to blink on instead of to the next state
160  m_state = (m_groupCounter > 0) ? STATE_BLINK_ON : STATE_BEGIN_GAP;
161  } else {
162  // this is the standard case, iterate to the next state
163  m_state = (PatternState)(m_state + 1);
164  }
165  // poor-mans recurse with the new state change (this transitions to a new state within the same tick)
166  goto replay;
