297 void set_terminal_nonblocking()
298 {
299  struct termios term_attr = {0};
300 
301  // Get the current terminal attributes and store them
302  tcgetattr(STDIN_FILENO, &term_attr);
303  orig_term_attr = term_attr;
304 
305  // Set the terminal in non-canonical mode (raw mode)
306  term_attr.c_lflag &= ~(ICANON | ECHO);
307 
308  // Set the minimum number of input bytes read at a time to 1
309  term_attr.c_cc[VMIN] = 1;
310 
311  // Set the timeout for read to 0 (no waiting)
312  term_attr.c_cc[VTIME] = 0;
313 
314  // Apply the new terminal attributes
315  tcsetattr(STDIN_FILENO, TCSANOW, &term_attr);
316 
317  // Set the terminal to non-blocking mode
318  int flags = fcntl(STDIN_FILENO, F_GETFL, 0);
319  fcntl(STDIN_FILENO, F_SETFL, flags | O_NONBLOCK);

680 // when the glove framework calls 'FastLED.show'
681 void VortexCLI::show()
682 {
683  if (!m_initialized) {
684  return;
685  }
686  if (m_outputType == OUTPUT_TYPE_SILENT) {
687  return;
688  }
689  string out;
690  get_terminal_size();
691  uint32_t wid = term_columns;
692  uint32_t odd = (wid) % 2;
693  uint32_t halfwid = wid / 2;
694  uint32_t midWid = (halfwid - ((2 + (m_outputType == OUTPUT_TYPE_HEX)) * LED_COUNT)) - 1;
695  if (m_inPlace) {
696  // this resets the cursor back to the beginning of the line and moves it up 12 lines
697  out += "\33[2K\033[17A\r";
698  // this is the top border line
699  out += "+";
700  out += LINE((wid + odd) - 2);
701  out += "+\n|";
702  // this is the left inner line
703  out += LINE(midWid + odd);
704  out += "=";
705  }
706  if (m_outputType == OUTPUT_TYPE_COLOR) {
707  // the color strip itself
708  for (uint32_t i = 0; i < m_numLeds; ++i) {
709  out += "\x1B[0m["; // opening |
710  out += "\x1B[48;2;"; // colorcode start
711  out += to_string(m_ledList[i].red) + ";"; // col red
712  out += to_string(m_ledList[i].green) + ";"; // col green
713  out += to_string(m_ledList[i].blue) + "m"; // col blue
714  out += " "; // colored space
715  out += "\x1B[0m]"; // ending |
716  }
717  } else if (m_outputType == OUTPUT_TYPE_HEX) {
718  // otherwise this just prints out the raw hex code if not in color mode
719  for (uint32_t i = 0; i < m_numLeds; ++i) {
720  char buf[128] = { 0 };
721  snprintf(buf, sizeof(buf), "%06X", m_ledList[i].raw());
722  out += buf;
723  }
724  }
725  if (!m_inPlace) {
726  out += "\n";
727  } else {
728  // the right inner line
729  out += "=";
730  out += LINE((midWid - 1) + odd);
731  // the end of middle line, fold and start of 3rd line
732  out += "|\n+";
733  out += LINE((wid + odd) - 2);
734  // space line after the box and before usage
735  out += "+\n";
736  out += SPACES(wid + odd);
737  // the usage message
738  for (uint32_t i = 0; i < NUM_USAGE; ++i) {
739  const char *brief = (wid < 70) ? input_usage_brief[i] : input_usage[i];
740  out += brief;
741  out += SPACES((wid + odd + 1) - strlen(brief));
742  }
743  }
744  printf("%s", out.c_str());
745  fflush(stdout);
746 }
