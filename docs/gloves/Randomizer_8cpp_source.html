148 Colorset Randomizer::rollColorset(Random &ctx)
149 {
150  Colorset randomSet;
151  // pick a random type of randomizer to use then use
152  // the randomizer to generate a random colorset
153  uint8_t randType = ctx.next8(0, 8);
154  switch (randType) {
155  default:
156  case 0:
157  randomSet.randomize(ctx);
158  break;
159  case 1:
160  randomSet.randomizeColors(ctx, 0, Colorset::ColorMode::THEORY);
161  break;
162  case 2:
163  randomSet.randomizeColors(ctx, 0, Colorset::ColorMode::MONOCHROMATIC);
164  break;
165  case 3:
166  randomSet.randomizeColors2(ctx, Colorset::ColorMode2::DOUBLE_SPLIT_COMPLIMENTARY);
167  break;
168  case 4:
169  randomSet.randomizeColors2(ctx, Colorset::ColorMode2::TETRADIC);
170  break;
171  case 5:
172  randomSet.randomize(ctx, 1);
173  break;
174  case 6:
175  randomSet.randomizeColors(ctx, 0, Colorset::ColorMode::EVENLY_SPACED);
176  break;
177  case 7:
178  randomSet.randomizeColors(ctx, 2, Colorset::ColorMode::EVENLY_SPACED);
179  break;
180  case 8:
181  randomSet.randomizeColors(ctx, 3, Colorset::ColorMode::EVENLY_SPACED);
182  break;
183  }
184 
185  // take the randomized colorset and repeate colors to make an additional pattern
186  if (randomSet.numColors() <= 4 && ctx.next8(0, 1) != 0) {
187  // using uint8_t here actually costs more bytes than int16_t
188  int16_t startingNumColors = randomSet.numColors();
189  for (int16_t i = 0; i < startingNumColors - 1; ++i) {
190  // add a duplicate of each color in the colorset to in reverse
191  randomSet.addColor(randomSet.get(startingNumColors - (i + 2)));
192  }
193  }
194  return randomSet;
195 }

198 {
