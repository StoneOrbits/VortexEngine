116 void FactoryReset::showReset()
117 {
118  // if we're on the thumb just set the rest to blank
119  if (m_curSelection == FINGER_THUMB) {
120  Leds::setRange(LED_FIRST, INDEX_TOP, RGB_WHITE0);
121  return;
122  }
123  // otherwise we're not on thumb, if the button isn't pressed
124  if (!g_pButton->isPressed()) {
125  // just idle blink from clear to blank
126  Leds::clearRange(LED_FIRST, INDEX_TOP);
127  Leds::blinkRange(LED_FIRST, INDEX_TOP, 250, 150, RGB_WHITE0);
128  return;
129  }
130 
131  // the button is pressed so show the reset countdown timer
132 
133  // the progress is how long the hold duration has been held
134  // relative to the factory reset threshold time
135  float progress = (float)g_pButton->holdDuration() / FACTORY_RESET_THRESHOLD_TICKS;
136  // the ledProgress is just an LED from pinky tip to index top based on progress
137  LedPos ledProgress = (LedPos)(progress * INDEX_TOP);
138  // max the led progress at index top (don't include thumb)
139  if (ledProgress > INDEX_TOP) {
140  // when we reach the end of the progress bar just blink white
141  Leds::blinkRange(LED_FIRST, INDEX_TOP, 80, 60, RGB_WHITE6);
142  return;
143  }
144 
145  // the off/on ms blink faster based on the progress
146  uint32_t offMs = 150 - (12 * ledProgress);
147  uint32_t onMs = 200 - (10 * ledProgress);
148  // the hue gets more red starting at 20 going in steps of 4 to min 0
149  int8_t hue = (ledProgress < 6) ? 20 - (4 * ledProgress) : 0;
150  // the 'endled' is the tip of the reset progress bar, since the progress
151  // bar starts full red and empties down to the pinky that means it is
152  // inverted from the 'ledProgress' which starts at 0 and grows
153  LedPos endLed = (LedPos)(INDEX_TOP - ledProgress);
154  // clear all the leds so that 'blinkRange' will blink from off to the designated color
155  Leds::clearRange(LED_FIRST, INDEX_TOP);
156  // blink to the calculated redish hue from pinky to the end led
157  Leds::blinkRange(LED_FIRST, endLed, offMs, onMs, HSVColor(hue, 255, 180));
158  // and blink the background the regular blank color
159  Leds::blinkRange((LedPos)(endLed + 1), INDEX_TOP, offMs, onMs, RGB_WHITE0);
