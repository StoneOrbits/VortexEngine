147 void VortexEngine::runMainLogic()
148 {
149  // the current tick
150  uint32_t now = Time::getCurtime();
151 
152  // if the menus are open and running then just return
153  if (Menus::run()) {
154  return;
155  }
156 
157  // check if we should enter the menu
158  if (g_pButton->isPressed() && g_pButton->holdDuration() > MENU_TRIGGER_THRESHOLD_TICKS) {
159  DEBUG_LOG("Entering Menu Selection...");
160  Menus::openMenuSelection();
161  return;
162  }
163 
164  // toggle auto cycle mode with many clicks at main modes
165  if (g_pButton->onConsecutivePresses(AUTO_CYCLE_MODES_CLICKS)) {
166  m_autoCycle = !m_autoCycle;
167  Leds::holdAll(m_autoCycle ? RGB_GREEN : RGB_RED);
168  }
169 
170  // if auto cycle is enabled and the last switch was more than the delay ago
171  if (m_autoCycle && (Modes::lastSwitchTime() + AUTO_RANDOM_DELAY < now)) {
172  // then switch to the next mode automatically
173  Modes::nextModeSkipEmpty();
174  }
175 
176  // otherwise just play the modes
177  Modes::play();
