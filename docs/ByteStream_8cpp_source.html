205 bool ByteStream::compress()
206 {
207  // only compress if we have valid data
208  if (!m_pData) {
209  DEBUG_LOG("No data to compress");
210  return false;
211  }
212  // recalculate the CRC even if we don't compress, this ensures the
213  // CRC is always up to date as long as compress() is called
214  recalcCRC();
215  // don't actually perform any compression if VORTEX SLIM is enabled
216 #if VORTEX_SLIM == 0
217  // check to see if the buffer is already compressed
218  if (is_compressed()) {
219  // already compressed
220  return true;
221  }
222  // use LZ4_compressBound for the output buffer size, it may be larger
223  // but it will allow for faster compression then we can shrink it after
224  ByteStream compressedBuffer(compress_size(m_pData->size));
225  // compress the data
226  int compressedSize = compress_buffer(m_pData->buf, compressedBuffer.m_pData->buf,
227  m_pData->size, compressedBuffer.m_capacity);
228  // check for compression error
229  if (compressedSize < 0) {
230  DEBUG_LOGF("Failed to compress, error: %d", compressedSize);
231  return false;
232  }
233  // check for useless compression
234  if (!compressedSize || (uint32_t)compressedSize >= m_pData->size) {
235  // cannot compress, or compressed no smaller than original
236  return true;
237  }
238  // update the buffer size
239  compressedBuffer.m_pData->size = compressedSize;
240  // buffer is now compressed
241  compressedBuffer.m_pData->flags = (m_pData->flags | BUFFER_FLAG_COMRPESSED);
242  // recalc the crc on the data buffer
243  compressedBuffer.m_pData->recalcCRC();
244  // shrink buffer capacity to match size
245  compressedBuffer.shrink();
246  // move into self
247  compressedBuffer.move(this);
248 #endif
249  return true;

252 bool ByteStream::decompress()
253 {
254  // ensure the internal buffer is sane after reading it out, this
255  // prevents segfaults if the internal size reports larger than capacity
256  sanity();
257  // only decompress if we have valid data
258  if (!m_pData || !m_pData->verify()) {
259  DEBUG_LOG("Cannot verify crc, not decompressing");
260  return false;
261  }
262  // don't actually perform any compression if VORTEX SLIM is enabled
263 #if VORTEX_SLIM == 0
264  if (!is_compressed()) {
265  // already decompressed
266  return true;
267  }
268  // first multiple will be 2x
269  uint32_t multiple = 1;
270  int decompressedSize = 0;
271  ByteStream decompressedBuffer;
272  do {
273  // double the amount each time
274  multiple *= 2;
275  // create buffer to receive decompressed data
276  decompressedBuffer.init(m_pData->size * multiple);
277  // decompress the data
278  decompressedSize = decompress_buffer(m_pData->buf, decompressedBuffer.m_pData->buf,
279  m_pData->size, decompressedBuffer.m_capacity);
280  } while (decompressedSize < 0 && multiple < 512);
281  // still error
282  if (decompressedSize < 0) {
283  ERROR_LOGF("Failed to decompress: %d", decompressedSize);
284  return false;
285  }
286  // size changed
287  decompressedBuffer.m_pData->size = decompressedSize;
288  // data is no longer compressed
289  decompressedBuffer.m_pData->flags = (m_pData->flags & ~BUFFER_FLAG_COMRPESSED);
290  // recalc crc of buffer
291  decompressedBuffer.m_pData->recalcCRC();
292  DEBUG_LOGF("Decompressed %u to %u bytes", m_pData->size, decompressedBuffer.m_pData->size);
293  // shrink buffer capacity to match size
294  decompressedBuffer.shrink();
295  // move into self
296  decompressedBuffer.move(this);
297 #endif
298  return true;
