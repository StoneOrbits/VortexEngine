103 bool Menus::runMenuSelection()
104 {
105  if (g_pButton->onShortClick()) {
106  // otherwise increment selection and wrap around at num menus
107  m_selection = (m_selection + 1) % NUM_MENUS;
108  DEBUG_LOGF("Cyling to ring menu %u", m_selection);
109  // reset the open time so that it starts again
110  m_openTime = Time::getCurtime();
111  // clear the leds
112  Leds::clearAll();
113  return true;
114  }
115  if (g_pButton2->onShortClick()) {
116  // increment down the menu list and wrap at 0
117  if (!m_selection) {
118  m_selection = MENU_COUNT - 1;
119  } else {
120  --m_selection;
121  }
122  Leds::clearAll();
123  return true;
124  }
125  // clear the leds so it always fills instead of replacing
126  Leds::clearAll();
127  // timings for blink later
128  uint8_t offtime = 200;
129  uint8_t ontime = 200;
130  // whether advanced menus are enabled
131  bool advMenus = Modes::advancedMenusEnabled();
132  // if the button was long pressed then select this menu, but we
133  // need to check the presstime to ensure we don't catch the initial
134  // release after opening the ringmenu
135  if (g_pButton->pressTime() >= m_openTime) {
136  // whether to open advanced menus or not
137  bool openAdv = (g_pButton->holdDuration() > ADV_MENU_DURATION_TICKS) && advMenus;
138  if (g_pButton->onLongClick()) {
139  // ringmenu is open so select the menu
140  DEBUG_LOGF("Selected ringmenu %s", menuList[m_selection].menuName);
141  // open the menu we have selected
142  if (!openMenu(m_selection, openAdv)) {
143  DEBUG_LOGF("Failed to initialize %s menu", menuList[m_selection].menuName);
144  return false;
145  }
146  // display the newly opened menu
147  return true;
148  }
149  // if you long click the 2nd button exit the menu list
150  if (g_pButton2->onLongClick()) {
151  // close the current menu when run returns false
152  closeCurMenu();
153  // return false to let the modes play
154  return false;
155  }
156  // if holding down to select the menu option
157  if (g_pButton->isPressed() && openAdv) {
158  // make it strobe aw yiss
159  offtime = HYPERSTROBE_OFF_DURATION;
160  ontime = HYPERSTROBE_ON_DURATION;
161  }
162  }
163  // blink every even/odd of every pair
164  for (Pair p = PAIR_FIRST; p < PAIR_COUNT; ++p) {
165  if (pairEven(p) < LED_COUNT) {
166  Leds::blinkIndex(pairEven(p), offtime, ontime, menuList[m_selection].color);
167  }
168  if (pairOdd(p) < LED_COUNT) {
169  Leds::setIndex(pairOdd(p), menuList[m_selection].color);
170  Leds::blinkIndex(pairOdd(p), offtime, ontime, RGB_OFF);
171  }
172  }
173  // check if the advanced menus have been enabled
174  if (g_pButton->onConsecutivePresses(ADVANCED_MENU_CLICKS)) {
175  // toggle the advanced menu
176  Modes::setAdvancedMenus(!advMenus);
177  // display a pink or red depending on whether the menu was enabled
178  for (uint16_t i = 0; i < 2; ++i) {
179  Leds::holdAll(advMenus ? RGB_RED : RGB_PINK);
180  }
181  }
182  // show when the user selects a menu option
183  showSelection(RGBColor(
184  menuList[m_selection].color.red << 3,
185  menuList[m_selection].color.green << 3,
186  menuList[m_selection].color.blue << 3));
187  // continue in the menu
188  return true;
189 }
