43 val RunTick(Vortex &vortex) {
44  // first run a tick
45  vortex.tick();
46  // then extract the color that was produced by the tick
47  RGBColor *leds = vortex.engine().leds().ledData();
48  val ledArray = val::array();
49  for (uint32_t i = 0; i < vortex.engine().leds().ledCount(); ++i) {
50  val color = val::object();
51  color.set("red", leds[i].red);
52  color.set("green", leds[i].green);
53  color.set("blue", leds[i].blue);
54  ledArray.set(i, color);
55  }
56  return ledArray;
57 }

764 void Vortex::handleNumber(char c)
765 {
766  if (!isdigit(c) || !m_lastCommand) {
767  return;
768  }
769  int number = c - '0';
770  char newc = 0;
771  // read the digits into the repeatAmount
772  while (1) {
773  newc = getchar();
774  if (!isdigit(newc)) {
775  // stop once we reach a non digit
776  break;
777  }
778  // accumulate the digits into the repeat amount
779  number = (number * 10) + (newc - '0');
780  }
781  // shove the last non-digit back into the stream
782  ungetc(newc, stdin);
783  DEBUG_LOGF("Repeating last command (%c) x%u times", m_lastCommand, number);
784  m_commandLog += to_string(number);
785  // check to see if we are repeating a 'rapid click' which is a special case
786  // because the rapid click command itself is composed or 'r' and a repeat count
787  // to designate how many rapid clicks to deliver
788  switch (m_lastCommand) {
789  case 'r':
790  // perform a rapid click number times
791  rapidClick(number);
792  break;
793  case 'b':
794  // select button index to target for click events
795  selectButton(number);
796  break;
797  default: // any other command
798  if (number > 0) {
799  // offset repeat amount by exactly 1 because it's already done
800  number--;
801  }
802  // repeat the last command that many times
803  while (number > 0) {
804  doCommand(m_lastCommand);
805  number--;
806  }
807  break;
808  }
809 }

811 // injects a command into the engine, the engine will parse one command
812 // per tick so multiple commands will be queued up
813 void Vortex::doCommand(char c)
814 {
815  if (!isprint(c)) {
816  return;
817  }
818 
819  m_commandLog += c;
820 
821  switch (c) {
822  case 'c':
823  if (m_lastCommand != c) {
824  DEBUG_LOG("Injecting short click");
825  }
826  shortClick();
827  break;
828  case 'l':
829  if (m_lastCommand != c) {
830  DEBUG_LOG("Injecting long click");
831  }
832  longClick();
833  break;
834  case 'm':
835  if (m_lastCommand != c) {
836  DEBUG_LOG("Injecting menu enter click");
837  }
838  menuEnterClick();
839  break;
840  case 'a':
841  if (m_lastCommand != c) {
842  DEBUG_LOG("Injecting adv menu enter click");
843  }
844  advMenuEnterClick();
845  break;
846  case 'd':
847  if (m_lastCommand != c) {
848  DEBUG_LOG("Injecting adv menu enter click");
849  }
850  deleteColClick();
851  break;
852  case 's':
853  if (m_lastCommand != c) {
854  DEBUG_LOG("Injecting sleep click");
855  }
856  sleepClick();
857  break;
858  case 'f':
859  if (m_lastCommand != c) {
860  DEBUG_LOG("Injecting force sleep click");
861  }
862  forceSleepClick();
863  break;
864  case 'q':
865  //case '\n':
866  if (m_lastCommand != c) {
867  DEBUG_LOG("Injecting quit click\n");
868  }
869  quitClick();
870  break;
871  case 't':
872  if (isButtonPressed()) {
873  if (m_lastCommand != c) {
874  DEBUG_LOG("Injecting release");
875  }
876  releaseButton();
877  } else {
878  if (m_lastCommand != c) {
879  DEBUG_LOG("Injecting press");
880  }
881  pressButton();
882  }
883  break;
884  case 'r':
885  // do nothing for the initial 'r', handleNumber() will handle the numeric
886  // repeat count that follows this letter and issue the rapidClick(amt) call
887  break;
888  case 'b':
889  // button selector, same as rapid click, it's a little weird because the number
890  // it not a repeat amount but actually the button index to select -- but none
891  // the less it works the same as rapid click so it will be handled inside handleNumber()
892  break;
893  case 'w':
894  if (m_lastCommand != c) {
895  DEBUG_LOG("Injecting wait");
896  }
897  sendWait();
898  break;
899  default:
900  handleNumber(c);
901  // return instead of break because this isn't a command
902  return;
903  }
904  m_lastCommand = c;
905 }

929 bool Vortex::tick()
930 {
931  if (!m_initialized) {
932  // do not cleanup prematurely here
933  return false;
934  }
935  // use ioctl to determine how many characters are on stdin so that
936  // we don't call getchar() too many times and accidentally block
937  uint32_t numInputs = getNumInputs();
938  if (m_lockstepEnabled && !numInputs) {
939  // don't tick till we have input
940  return true;
941  }
942  // iterate the number of inputs on stdin and parse each letter
943  // into a command for the engine, this will inject all of the commands
944  // that are available into the engine but that doesn't necessarily
945  // mean that the engine will do anything with them right away
946  for (uint32_t i = 0; i < numInputs; ++i) {
947  doCommand(getchar());
948  }
949  // tick the vortex engine forward
950  m_engine.tick();
951  return true;
952 }

1721 void Vortex::handleInputQueue(Button *buttons, uint32_t numButtons)
1722 {
1723  // if there's nothing in the queue just return
1724  if (!m_buttonEventQueue.size()) {
1725  return;
1726  }
1727  // pop the event from the front of the queue
1728  VortexButtonEvent buttonEvent = m_buttonEventQueue.front();
1729  m_buttonEventQueue.pop_front();
1730  // the target button for this event (no target if event is just 'wait')
1731  Button *pButton = buttons;
1732  if (buttonEvent.type != EVENT_WAIT && buttonEvent.type != EVENT_RAPID_CLICK) {
1733  // make sure the button that is targeted is actually a valid index
1734  if (buttonEvent.target >= numButtons) {
1735  return;
1736  }
1737  // assigned the button based on the array index target
1738  pButton = buttons + buttonEvent.target;
1739  }
1740  // switch on the type of event and then run the operation
1741  switch (buttonEvent.type) {
1742  case EVENT_NONE:
1743  default:
1744  break;
1745  case EVENT_SHORT_CLICK:
1746  pButton->m_newRelease = true;
1747  pButton->m_shortClick = true;
1748  pButton->m_pressTime = m_engine.time().getCurtime();
1749  pButton->m_holdDuration = SHORT_CLICK_THRESHOLD_TICKS - 1;
1750  DEBUG_LOG("Injecting short click");
1751  break;
1752  case EVENT_LONG_CLICK:
1753  pButton->m_newRelease = true;
1754  pButton->m_longClick = true;
1755  pButton->m_pressTime = m_engine.time().getCurtime();
1756  pButton->m_holdDuration = SHORT_CLICK_THRESHOLD_TICKS + 1;
1757  DEBUG_LOG("Injecting long click");
1758  break;
1759  case EVENT_MENU_ENTER_CLICK:
1760  // to do this we simply press the button and set the press time
1761  // to something more than the menu trigger threshold that will make
