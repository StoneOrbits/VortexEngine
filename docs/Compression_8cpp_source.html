260 static unsigned LZ4_NbCommonBytes(reg_t val)
261 {
262  assert(val != 0);
263  if (LZ4_isLittleEndian()) {
264  if (sizeof(val) == 8) {
265  /*-*************************************************************************************************
266  * ARM64EC is a Microsoft-designed ARM64 ABI compatible with AMD64 applications on ARM64 Windows 11.
267  * The ARM64EC ABI does not support AVX/AVX2/AVX512 instructions, nor their relevant intrinsics
268  * including _tzcnt_u64. Therefore, we need to neuter the _tzcnt_u64 code path for ARM64EC.
269  ****************************************************************************************************/
270  const U64 m = 0x0101010101010101ULL;
271  val ^= val - 1;
272  return (unsigned)(((U64)((val & (m - 1)) * m)) >> 56);
273  } else /* 32 bits */ {
274  const U32 m = 0x01010101;
275  return (unsigned)((((val - 1) ^ val) & (m - 1)) * m) >> 24;
276  }
277  } else /* Big Endian CPU */ {
278  if (sizeof(val) == 8) {
279  /* this method doesn't consume memory space like the previous one,
280  * but it contains several branches,
281  * that may end up slowing execution */
282  static const U32 by32 = sizeof(val) * 4; /* 32 on 64 bits (goal), 16 on 32 bits.
283  Just to avoid some static analyzer complaining about shift by 32 on 32-bits target.
284  Note that this code path is never triggered in 32-bits mode. */
285  unsigned r;
286  if (!(val >> by32)) { r = 4; } else { r = 0; val >>= by32; }
287  if (!(val >> 16)) { r += 2; val >>= 8; } else { val >>= 24; }
288  r += (!val);
289  return r;
290  } else /* 32 bits */ {
291  val >>= 8;
292  val = ((((val + 0x00FFFF00) | 0x00FFFFFF) + val) |
293  (val + 0x00FF0000)) >> 24;
294  return (unsigned)val ^ 3;
295  }
296  }

618  } else if (dictDirective == usingExtDict) {

620  DEBUGLOG(7, "extDict candidate: matchIndex=%5u < startIndex=%5u", matchIndex, startIndex);
621  assert(startIndex - matchIndex >= MINMATCH);
622  assert(dictBase);
623  match = dictBase + matchIndex;
624  lowLimit = dictionary;
625  } else {
626  match = base + matchIndex;
627  lowLimit = (const BYTE *)source;
628  }
629  } else { /* single continuous memory segment */
630  match = base + matchIndex;
631  }
632  forwardH = LZ4_hashPosition(forwardIp, tableType);
633  LZ4_putIndexOnHash(current, h, cctx->hashTable, tableType);
634 
635  DEBUGLOG(7, "candidate at pos=%u (offset=%u \n", matchIndex, current - matchIndex);
636  if ((dictIssue == dictSmall) && (matchIndex < prefixIdxLimit)) { continue; } /* match outside of valid area */
637  assert(matchIndex < current);
638  if (((tableType != byU16) || (LZ4_DISTANCE_MAX < LZ4_DISTANCE_ABSOLUTE_MAX))
639  && (matchIndex + LZ4_DISTANCE_MAX < current)) {
640  continue;
641  } /* too far */
642  assert((current - matchIndex) <= LZ4_DISTANCE_MAX); /* match now expected within distance */
643 
644  if (LZ4_read32(match) == LZ4_read32(ip)) {
645  if (maybe_extMem) offset = current - matchIndex;
646  break; /* match found */
647  }
648 
649  } while (1);
650  }
651 
652  /* Catch up */
653  filledIp = ip;
654  while (((ip > anchor) & (match > lowLimit)) && (unlikely(ip[-1] == match[-1]))) { ip--; match--; }
655 
656  /* Encode Literals */
657  {
658  unsigned const litLength = (unsigned)(ip - anchor);
659  token = op++;
660  if ((outputDirective == limitedOutput) && /* Check output buffer overflow */
661  (unlikely(op + litLength + (2 + 1 + LASTLITERALS) + (litLength / 255) > olimit))) {
662  return 0; /* cannot compress within `dst` budget. Stored indexes in hash table are nonetheless fine */
663  }
664  if ((outputDirective == fillOutput) &&
665  (unlikely(op + (litLength + 240) / 255 /* litlen */ + litLength /* literals */ + 2 /* offset */ + 1 /* token */ + MFLIMIT - MINMATCH /* min last literals so last match is <= end - MFLIMIT */ > olimit))) {
666  op--;
667  goto _last_literals;
668  }
669  if (litLength >= RUN_MASK) {
670  int len = (int)(litLength - RUN_MASK);
671  *token = (RUN_MASK << ML_BITS);
672  for (; len >= 255; len -= 255) *op++ = 255;
673  *op++ = (BYTE)len;
674  } else *token = (BYTE)(litLength << ML_BITS);
675 
676  /* Copy Literals */
677  LZ4_wildCopy8(op, anchor, op + litLength);
678  op += litLength;
679  DEBUGLOG(6, "seq.start:%i, literals=%u, match.start:%i",
680  (int)(anchor - (const BYTE *)source), litLength, (int)(ip - (const BYTE *)source));
681  }
682 
683  _next_match:
684  /* at this stage, the following variables must be correctly set :
685  * - ip : at start of LZ operation
686  * - match : at start of previous pattern occurrence; can be within current prefix, or within extDict
687  * - offset : if maybe_ext_memSegment==1 (constant)
688  * - lowLimit : must be == dictionary to mean "match is within extDict"; must be == source otherwise
689  * - token and *token : position to write 4-bits for match length; higher 4-bits for literal length supposed already written
690  */
691 
692  if ((outputDirective == fillOutput) &&
693  (op + 2 /* offset */ + 1 /* token */ + MFLIMIT - MINMATCH /* min last literals so last match is <= end - MFLIMIT */ > olimit)) {
694  /* the match was too close to the end, rewind and go to last literals */
695  op = token;
696  goto _last_literals;
697  }
698 
699  /* Encode Offset */
700  if (maybe_extMem) { /* static test */
701  DEBUGLOG(6, " with offset=%u (ext if > %i)", offset, (int)(ip - (const BYTE *)source));
702  assert(offset <= LZ4_DISTANCE_MAX && offset > 0);
703  LZ4_writeLE16(op, (U16)offset); op += 2;
704  } else {
705  DEBUGLOG(6, " with offset=%u (same segment)", (U32)(ip - match));
706  assert(ip - match <= LZ4_DISTANCE_MAX);
707  LZ4_writeLE16(op, (U16)(ip - match)); op += 2;
708  }
709 
710  /* Encode MatchLength */
711  {
712  unsigned matchCode;
713 
714  if ((dictDirective == usingExtDict || dictDirective == usingDictCtx)
715  && (lowLimit == dictionary) /* match within extDict */) {
716  const BYTE *limit = ip + (dictEnd - match);
717  assert(dictEnd > match);
718  if (limit > matchlimit) limit = matchlimit;
719  matchCode = LZ4_count(ip + MINMATCH, match + MINMATCH, limit);
720  ip += (size_t)matchCode + MINMATCH;
721  if (ip == limit) {
722  unsigned const more = LZ4_count(limit, (const BYTE *)source, matchlimit);
723  matchCode += more;
724  ip += more;
725  }
726  DEBUGLOG(6, " with matchLength=%u starting in extDict", matchCode + MINMATCH);
727  } else {
728  matchCode = LZ4_count(ip + MINMATCH, match + MINMATCH, matchlimit);
729  ip += (size_t)matchCode + MINMATCH;
730  DEBUGLOG(6, " with matchLength=%u", matchCode + MINMATCH);
731  }
732 
733  if ((outputDirective) && /* Check output buffer overflow */
734  (unlikely(op + (1 + LASTLITERALS) + (matchCode + 240) / 255 > olimit))) {
735  if (outputDirective == fillOutput) {
736  /* Match description too long : reduce it */
737  U32 newMatchCode = 15 /* in token */ - 1 /* to avoid needing a zero byte */ + ((U32)(olimit - op) - 1 - LASTLITERALS) * 255;
738  ip -= matchCode - newMatchCode;
739  assert(newMatchCode < matchCode);
740  matchCode = newMatchCode;
741  if (unlikely(ip <= filledIp)) {
742  /* We have already filled up to filledIp so if ip ends up less than filledIp
743  * we have positions in the hash table beyond the current position. This is
744  * a problem if we reuse the hash table. So we have to remove these positions
745  * from the hash table.
746  */
747  const BYTE *ptr;
748  DEBUGLOG(5, "Clearing %u positions", (U32)(filledIp - ip));
749  for (ptr = ip; ptr <= filledIp; ++ptr) {
750  U32 const h = LZ4_hashPosition(ptr, tableType);
751  LZ4_clearHash(h, cctx->hashTable, tableType);
752  }
753  }
754  } else {
755  assert(outputDirective == limitedOutput);
756  return 0; /* cannot compress within `dst` budget. Stored indexes in hash table are nonetheless fine */
757  }
758  }
759  if (matchCode >= ML_MASK) {
760  *token += ML_MASK;
761  matchCode -= ML_MASK;
762  LZ4_write32(op, 0xFFFFFFFF);
763  while (matchCode >= 4 * 255) {
764  op += 4;
765  LZ4_write32(op, 0xFFFFFFFF);
766  matchCode -= 4 * 255;
767  }
768  op += matchCode / 255;
769  *op++ = (BYTE)(matchCode % 255);
770  } else
771  *token += (BYTE)(matchCode);
772  }
773  /* Ensure we have enough space for the last literals. */
774  assert(!(outputDirective == fillOutput && op + 1 + LASTLITERALS > olimit));
775 
776  anchor = ip;
777 
778  /* Test end of chunk */
779  if (ip >= mflimitPlusOne) break;
780 
781  /* Fill table */
782  LZ4_putPosition(ip - 2, cctx->hashTable, tableType, base);
783 
784  /* Test next position */
785  if (tableType == byPtr) {
786 
787  match = LZ4_getPosition(ip, cctx->hashTable, tableType, base);
788  LZ4_putPosition(ip, cctx->hashTable, tableType, base);
789  if ((match + LZ4_DISTANCE_MAX >= ip)
790  && (LZ4_read32(match) == LZ4_read32(ip))) {
791  token = op++; *token = 0; goto _next_match;
792  }
793 
794  } else { /* byU32, byU16 */
795 
796  U32 const h = LZ4_hashPosition(ip, tableType);
797  U32 const current = (U32)(ip - base);
798  U32 matchIndex = LZ4_getIndexOnHash(h, cctx->hashTable, tableType);
799  assert(matchIndex < current);
800  if (dictDirective == usingDictCtx) {
801  if (matchIndex < startIndex) {
802  /* there was no match, try the dictionary */
803  matchIndex = LZ4_getIndexOnHash(h, dictCtx->hashTable, byU32);
804  match = dictBase + matchIndex;
805  lowLimit = dictionary; /* required for match length counter */
806  matchIndex += dictDelta;
807  } else {
808  match = base + matchIndex;
809  lowLimit = (const BYTE *)source; /* required for match length counter */
810  }
811  } else if (dictDirective == usingExtDict) {
812  if (matchIndex < startIndex) {
813  assert(dictBase);
814  match = dictBase + matchIndex;
815  lowLimit = dictionary; /* required for match length counter */
816  } else {
817  match = base + matchIndex;
818  lowLimit = (const BYTE *)source; /* required for match length counter */
819  }
820  } else { /* single memory segment */
821  match = base + matchIndex;
822  }
823  LZ4_putIndexOnHash(current, h, cctx->hashTable, tableType);
824  assert(matchIndex < current);
825  if (((dictIssue == dictSmall) ? (matchIndex >= prefixIdxLimit) : 1)
826  && (((tableType == byU16) && (LZ4_DISTANCE_MAX == LZ4_DISTANCE_ABSOLUTE_MAX)) ? 1 : (matchIndex + LZ4_DISTANCE_MAX >= current))
827  && (LZ4_read32(match) == LZ4_read32(ip))) {
828  token = op++;
829  *token = 0;
830  if (maybe_extMem) offset = current - matchIndex;
831  DEBUGLOG(6, "seq.start:%i, literals=%u, match.start:%i",
832  (int)(anchor - (const BYTE *)source), 0, (int)(ip - (const BYTE *)source));
833  goto _next_match;
834  }
835  }
836 
837  /* Prepare next loop */
838  forwardH = LZ4_hashPosition(++ip, tableType);
839 
840  }
841 
842 _last_literals:
843  /* Encode Last Literals */
844  {
845  size_t lastRun = (size_t)(iend - anchor);
846  if ((outputDirective) && /* Check output buffer overflow */
847  (op + lastRun + 1 + ((lastRun + 255 - RUN_MASK) / 255) > olimit)) {
848  if (outputDirective == fillOutput) {
849  /* adapt lastRun to fill 'dst' */
850  assert(olimit >= op);
851  lastRun = (size_t)(olimit - op) - 1/*token*/;
852  lastRun -= (lastRun + 256 - RUN_MASK) / 256; /*additional length tokens*/
853  } else {
854  assert(outputDirective == limitedOutput);
855  return 0; /* cannot compress within `dst` budget. Stored indexes in hash table are nonetheless fine */
856  }
857  }
858  DEBUGLOG(6, "Final literal run : %i literals", (int)lastRun);
859  if (lastRun >= RUN_MASK) {
860  size_t accumulator = lastRun - RUN_MASK;
861  *op++ = RUN_MASK << ML_BITS;
862  for (; accumulator >= 255; accumulator -= 255) *op++ = 255;
863  *op++ = (BYTE)accumulator;
864  } else {
865  *op++ = (BYTE)(lastRun << ML_BITS);
866  }
867  LZ4_memcpy(op, anchor, lastRun);
868  ip = anchor + lastRun;
869  op += lastRun;
870  }
871 
872  if (outputDirective == fillOutput) {
873  *inputConsumed = (int)(((const char *)ip) - source);
874  }
875  result = (int)(((char *)op) - dest);
876  assert(result > 0);
877  DEBUGLOG(5, "LZ4_compress_generic: compressed %i bytes into %i bytes", inputSize, result);
878  return result;
879 }

989 typedef enum { endOnOutputSize = 0, endOnInputSize = 1 } endCondition_directive;

995 /* Read the variable-length literal or match length.
996  *
997  * ip - pointer to use as input.
998  * lencheck - end ip. Return an error if ip advances >= lencheck.
999  * loop_check - check ip >= lencheck in body of loop. Returns loop_error if so.
1000  * initial_check - check ip >= lencheck before start of loop. Returns initial_error if so.
1001  * error (output) - error code. Should be set to 0 before call.
1002  */
1003 typedef enum { loop_error = -2, initial_error = -1, ok = 0 } variable_length_error;
1004 unsigned read_variable_length(const BYTE **ip, const BYTE *lencheck,
1005  int loop_check, int initial_check,
1006  variable_length_error *error)
1007 {
1008  U32 length = 0;
1009  U32 s;
1010  if (initial_check && unlikely((*ip) >= lencheck)) { /* overflow detection */
1011  *error = initial_error;
1012  return length;
1013  }
1014  do {
1015  s = **ip;
1016  (*ip)++;
1017  length += s;
1018  if (loop_check && unlikely((*ip) >= lencheck)) { /* overflow detection */
1019  *error = loop_error;
1020  return length;
1021  }
1022  } while (s == 255);
1023 
1024  return length;
1025 }

1033 int LZ4_decompress_generic(
1034  const char *const src,
1035  char *const dst,
1036  int srcSize,
1037  int outputSize, /* If endOnInput==endOnInputSize, this value is `dstCapacity` */
1038 
1039  endCondition_directive endOnInput, /* endOnOutputSize, endOnInputSize */
1040  earlyEnd_directive partialDecoding, /* full, partial */
1041  dict_directive dict, /* noDict, withPrefix64k, usingExtDict */
1042  const BYTE *const lowPrefix, /* always <= dst, == dst when no prefix */
1043  const BYTE *const dictStart, /* only if dict==usingExtDict */
1044  const size_t dictSize /* note : = 0 if noDict */
1045 )
1046 {
1047  if ((src == NULL) || (outputSize < 0)) { return -1; }
1048 
1049  {
1050  const BYTE *ip = (const BYTE *)src;
1051  const BYTE *const iend = ip + srcSize;
1052 
1053  BYTE *op = (BYTE *)dst;
1054  BYTE *const oend = op + outputSize;
1055  BYTE *cpy;
1056 
1057  const BYTE *const dictEnd = (dictStart == NULL) ? NULL : dictStart + dictSize;
1058 
1059  const int safeDecode = (endOnInput == endOnInputSize);
1060  const int checkOffset = ((safeDecode) && (dictSize < (int)(64 KB)));
1061 
1062 
1063  /* Set up the "end" pointers for the shortcut. */
1064  const BYTE *const shortiend = iend - (endOnInput ? 14 : 8) /*maxLL*/ - 2 /*offset*/;
1065  const BYTE *const shortoend = oend - (endOnInput ? 14 : 8) /*maxLL*/ - 18 /*maxML*/;
1066 
1067  const BYTE *match;
1068  size_t offset;
1069  unsigned token;
1070  size_t length;
1071 
1072 
1073  DEBUGLOG(5, "LZ4_decompress_generic (srcSize:%i, dstSize:%i)", srcSize, outputSize);
1074 
1075  /* Special cases */
1076  assert(lowPrefix <= op);
1077  if ((endOnInput) && (unlikely(outputSize == 0))) {
1078  /* Empty output buffer */
1079  if (partialDecoding) return 0;
1080  return ((srcSize == 1) && (*ip == 0)) ? 0 : -1;
1081  }
1082  if ((!endOnInput) && (unlikely(outputSize == 0))) { return (*ip == 0 ? 1 : -1); }
1083  if ((endOnInput) && unlikely(srcSize == 0)) { return -1; }
1084 
1085  /* Currently the fast loop shows a regression on qualcomm arm chips. */
1086  if ((oend - op) < FASTLOOP_SAFE_DISTANCE) {
1087  DEBUGLOG(6, "skip fast decode loop");
1088  goto safe_decode;
1089  }
1090 
1091  /* Fast loop : decode sequences as long as output < iend-FASTLOOP_SAFE_DISTANCE */
1092  while (1) {
1093  /* Main fastloop assertion: We can always wildcopy FASTLOOP_SAFE_DISTANCE */
1094  assert(oend - op >= FASTLOOP_SAFE_DISTANCE);
1095  if (endOnInput) { assert(ip < iend); }
1096  token = *ip++;
1097  length = token >> ML_BITS; /* literal length */
1098 
1099  assert(!endOnInput || ip <= iend); /* ip < iend before the increment */
1100 
1101  /* decode literal length */
1102  if (length == RUN_MASK) {
1103  variable_length_error error = ok;
1104  length += read_variable_length(&ip, iend - RUN_MASK, (int)endOnInput, (int)endOnInput, &error);
1105  if (error == initial_error) { goto _output_error; }
1106  if ((safeDecode) && unlikely((uptrval)(op)+length < (uptrval)(op))) { goto _output_error; } /* overflow detection */
1107  if ((safeDecode) && unlikely((uptrval)(ip)+length < (uptrval)(ip))) { goto _output_error; } /* overflow detection */
1108 
1109  /* copy literals */
1110  cpy = op + length;
1111  LZ4_STATIC_ASSERT(MFLIMIT >= WILDCOPYLENGTH);
1112  if (endOnInput) { /* LZ4_decompress_safe() */
1113  if ((cpy > oend - 32) || (ip + length > iend - 32)) { goto safe_literal_copy; }
1114  LZ4_wildCopy32(op, ip, cpy);
1115  } else { /* LZ4_decompress_fast() */
1116  if (cpy > oend - 8) { goto safe_literal_copy; }
1117  LZ4_wildCopy8(op, ip, cpy); /* LZ4_decompress_fast() cannot copy more than 8 bytes at a time :
1118  * it doesn't know input length, and only relies on end-of-block properties */
1119  }
1120  ip += length; op = cpy;
1121  } else {
1122  cpy = op + length;
1123  if (endOnInput) { /* LZ4_decompress_safe() */
1124  DEBUGLOG(7, "copy %u bytes in a 16-bytes stripe", (unsigned)length);
1125  /* We don't need to check oend, since we check it once for each loop below */
1126  if (ip > iend - (16 + 1/*max lit + offset + nextToken*/)) { goto safe_literal_copy; }
1127  /* Literals can only be 14, but hope compilers optimize if we copy by a register size */
1128  LZ4_memcpy(op, ip, 16);
1129  } else { /* LZ4_decompress_fast() */
1130  /* LZ4_decompress_fast() cannot copy more than 8 bytes at a time :
1131  * it doesn't know input length, and relies on end-of-block properties */
1132  LZ4_memcpy(op, ip, 8);
1133  if (length > 8) { LZ4_memcpy(op + 8, ip + 8, 8); }
1134  }
1135  ip += length; op = cpy;
1136  }
1137 
1138  /* get offset */
1139  offset = LZ4_readLE16(ip); ip += 2;
1140  match = op - offset;
1141  assert(match <= op);
1142 
1143  /* get matchlength */
1144  length = token & ML_MASK;
1145 
1146  if (length == ML_MASK) {
1147  variable_length_error error = ok;
1148  if ((checkOffset) && (unlikely(match + dictSize < lowPrefix))) { goto _output_error; } /* Error : offset outside buffers */
1149  length += read_variable_length(&ip, iend - LASTLITERALS + 1, (int)endOnInput, 0, &error);
1150  if (error != ok) { goto _output_error; }
1151  if ((safeDecode) && unlikely((uptrval)(op)+length < (uptrval)op)) { goto _output_error; } /* overflow detection */
1152  length += MINMATCH;
1153  if (op + length >= oend - FASTLOOP_SAFE_DISTANCE) {
1154  goto safe_match_copy;
1155  }
1156  } else {
1157  length += MINMATCH;
1158  if (op + length >= oend - FASTLOOP_SAFE_DISTANCE) {
1159  goto safe_match_copy;
1160  }
