143 void ZigzagPattern::Snake::drawSnake()
144 {
145  uint8_t segment_position;
146  int32_t colIndex = 0;
147  for (uint8_t segment = 0; segment < m_snakeSize; ++segment) {
148  // calculate the position of this segment, if the segment being rendered is
149  // greater than the head position it means the snake head is at an index that
150  // won't support it's entire tail without wrapping below 0
151  if (segment > m_step) {
152  segment_position = NUM_ZIGZAG_STEPS - (segment - m_step);
153  } else {
154  segment_position = m_step - segment;
155  }
156  // adjust the brightness of the color before setting
157  RGBColor col = m_colorset.peek(colIndex);
158  if (segment < m_snakeSize) {
159  col.adjustBrightness(m_fadeAmount * segment);
160  }
161  // lookup the target in the step positions array and turn it on with given color/brightness
162  Leds::setIndex(ledStepPositions[segment_position], col);
163  // if this segment is on the step where the color changes
164  if (segment_position == m_changeBoundary) {
165  // then decrement the color index for the rest of the snake so that the
166  // tail of the snake has the old color till it reaches the change point
167  colIndex--;
168  }
169  }
