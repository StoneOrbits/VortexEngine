149 void ColorSelect::onLongClick()
150 {
151  if (m_advanced) {
152  Colorset set = m_previewMode.getColorset();
153  set.set(0, set.get(1));
154  m_previewMode.setColorset(set);
155  return;
156  }
157  // if we're on 'exit' and we're on any menu past the slot selection
158  if (m_curSelection == 4 && m_state > STATE_PICK_SLOT) {
159  // move back to the previous selection
160  m_state = (ColorSelectState)(m_state - 1);
161  // if we're back to the slot selection then set our position back to the
162  // slot we selected, otherwise it's more visually appealing to just return
163  // to the first selection on the previous menu
164  m_curSelection = (m_state == STATE_PICK_SLOT) ? m_targetSlot : 0;
165  return;
166  }
167  // reuse these variables lots
168  uint8_t numColors = m_colorset.numColors();
169  uint32_t holdDur = g_pButton->holdDuration();
170  switch (m_state) {
171  case STATE_INIT:
172  // nothing
173  return;
174  case STATE_PICK_SLOT:
175  // if the exit is selected then save and exit, this depends on the number
176  // of colors in the colorset. If the colorset is full then the exit is
177  // just the number of colors (8) but if it's not full then the exit is the
178  // number of colors + 1. Example: with 4 cols, cols are on 0, 1, 2, 3,
179  // add-color is 4, and exit is 5
180  if (m_curSelection == numColors + (numColors < MAX_COLOR_SLOTS)) {
181  Mode *cur = Modes::curMode();
182  cur->setColorsetMap(m_targetLeds, m_colorset);
183  cur->init();
184  leaveMenu(true);
185  return;
186  }
187  // handle if user releases during the delete option
188  if (m_curSelection < numColors &&
189  holdDur >= DELETE_THRESHOLD_TICKS &&
190  (holdDur % (DELETE_CYCLE_TICKS * 2)) > (DELETE_CYCLE_TICKS)) {
191  // delete current slot
192  m_colorset.removeColor(m_curSelection);
193  if (m_curSelection > numColors) {
194  m_curSelection--;
195  }
196  return;
197  }
198  // otherwise store the target slot continue onto the hue selection
199  m_targetSlot = m_curSelection;
200  break;
201  case STATE_PICK_HUE1:
202  m_targetHue1 = m_curSelection;
203  m_newColor.hue = m_targetHue1 * (255 / 4);
204  break;
205  case STATE_PICK_HUE2:
206  m_newColor.hue = (m_targetHue1 * (255 / 4)) + m_curSelection * (255 / 16);
207  break;
208  case STATE_PICK_SAT:
209  m_newColor.sat = sats[m_curSelection];
210  break;
211  case STATE_PICK_VAL:
212  // no m_targetVal because you can't go back after this
213  m_newColor.val = vals[m_curSelection];
214  // specifically using hsv to rgb rainbow to generate the color
215  m_colorset.set(m_targetSlot, m_newColor);
216  m_curSelection = m_targetSlot;
217  m_state = STATE_PICK_SLOT;
218  return;
219  }
220  m_state = (ColorSelectState)(m_state + 1);
221  m_curSelection = 0;
222 }

224 void ColorSelect::showSlotSelection()
225 {
226  uint8_t exitIndex = m_colorset.numColors();
227  uint32_t holdDur = g_pButton->holdDuration();
228  bool withinNumColors = m_curSelection < exitIndex;
229  bool holdDurationCheck = g_pButton->isPressed() && holdDur >= DELETE_THRESHOLD_TICKS;
230  bool holdDurationModCheck = (holdDur % (DELETE_CYCLE_TICKS * 2)) > DELETE_CYCLE_TICKS;
231  const RGBColor &col = m_colorset[m_curSelection];
232  if (withinNumColors && holdDurationCheck && holdDurationModCheck) {
233  // breath red for delete slot
234  Leds::blinkIndex(LED_0, 50, 100, col);
235  Leds::breathIndex(LED_1, 0, holdDur);
236  } else if (withinNumColors) {
237  if (col.empty()) {
238  Leds::setIndex(LED_0, RGB_WHITE0);
239  }
240  Leds::blinkAll(150, 650, col);
241  } else if (exitIndex < MAX_COLOR_SLOTS) {
242  if (m_curSelection == exitIndex) {
243  // blink both leds and blink faster to indicate 'add' new color
244  Leds::blinkAll(100, 150, RGB_WHITE2);
245  }
246  exitIndex++;
247  }
248  if (m_curSelection == exitIndex) {
249  // display the full set
250  showFullSet(50, 100);
251  // set LED_1 to green to indicate save and exit
252  Leds::setIndex(LED_1, RGB_GREEN2);
253  // if not on exitIndex or add new color set LED_1 based on button state
254  } else if (m_curSelection != m_colorset.numColors() && !holdDurationCheck) {
255  Leds::setIndex(LED_1, g_pButton->isPressed() ? RGB_OFF : RGB_WHITE2);
256  }

277  sat = 255;
