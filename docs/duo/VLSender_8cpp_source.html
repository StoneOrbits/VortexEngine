76 bool VLSender::send()
77 {
78  // must have at least one block to send
79  if (m_numBlocks < 1) {
80  m_isSending = false;
81  return m_isSending;
82  }
83  if (!m_isSending) {
84  beginSend();
85  }
86  if (m_lastSendTime > 0 && m_lastSendTime + VL_DEFAULT_BLOCK_SPACING > Time::getCurtime()) {
87  // don't send yet
88  return m_isSending;
89  }
90  // blocksize is default unless it's the last block, then it's the remainder
91  uint32_t blocksize = (m_numBlocks == 1) ? m_remainder : VL_DEFAULT_BLOCK_SIZE;
92  DEBUG_LOGF("VLSender Sending block #%u", m_numBlocks);
93  // pointer to the data at the correct location
94  const uint8_t *buf_ptr = m_bitStream.data() + m_writeCounter;
95  // iterate each byte in the block and write it
96  for (uint32_t i = 0; i < blocksize; ++i) {
97  // write the byte
98  sendByte(*buf_ptr);
99  // increment to next byte in buffer
100  buf_ptr++;
101  // record the written byte
102  m_writeCounter++;
103  }
104  // wrote one block
105  m_numBlocks--;
106  // still sending if we have more blocks
107  m_isSending = (m_numBlocks > 0);

109  m_lastSendTime = Time::getCurtime();

111  return m_isSending;

134 {
