142  setKeychainModeState((keychain_mode_state)(m_keychain_modeState + 1));
143  return MENU_CONTINUE;

145  // as long as the button is held down the leds clear
146  if (g_pButton->isPressed()) {

149  }
150  // check for sleep after some seconds

152  // Optional: indicate when the keychain is turning off (useful mainly for debugging)

160  Leds::clearAll();
