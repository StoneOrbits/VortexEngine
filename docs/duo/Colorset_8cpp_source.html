242 void Colorset::randomizeColors(Random &ctx, uint8_t numColors, ColorMode mode)
243 {
244  clear();
245  if (!numColors) {
246  numColors = ctx.next8(mode == MONOCHROMATIC ? 2 : 1, 9);
247  }
248  uint8_t randomizedHue = ctx.next8();
249  uint8_t colorGap = 0;
250  if (mode == THEORY && numColors > 1) {
251  colorGap = ctx.next8(16, 256 / (numColors - 1));
252  }
253  ValueStyle valStyle = (ValueStyle)ctx.next8(0, VAL_STYLE_COUNT);
254  // the doubleStyle decides if some colors are added to the set twice
255  uint8_t doubleStyle = 0;
256  if (numColors <= 7) {
257  doubleStyle = (ctx.next8(0, 1));
258  }
259  if (numColors <= 4) {
260  doubleStyle = (ctx.next8(0, 2));
261  }
262  for (uint8_t i = 0; i < numColors; i++) {
263  uint8_t hueToUse;
264  uint8_t valueToUse = 255;
265  if (mode == THEORY) {
266  hueToUse = (randomizedHue + (i * colorGap));
267  } else if (mode == MONOCHROMATIC) {
268  hueToUse = randomizedHue;
269  valueToUse = 255 - (i * (256 / numColors));
270  } else { // EVENLY_SPACED
271  hueToUse = (randomizedHue + (256 / numColors) * i);
272  }
273  addColorWithValueStyle(ctx, hueToUse, valueToUse, valStyle, numColors, i);
274  // double all colors or only first color
275  if (doubleStyle == 2 || (doubleStyle == 1 && !i)) {
276  addColorWithValueStyle(ctx, hueToUse, valueToUse, valStyle, numColors, i);
277  }
