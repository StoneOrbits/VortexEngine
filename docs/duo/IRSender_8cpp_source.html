79 bool IRSender::send()
80 {
81  // must have at least one block to send
82  if (m_numBlocks < 1) {
83  m_isSending = false;
84  return m_isSending;
85  }
86  if (!m_isSending) {
87  beginSend();
88  }
89  if (m_lastSendTime > 0 && m_lastSendTime + IR_DEFAULT_BLOCK_SPACING > Time::getCurtime()) {
90  // don't send yet
91  return m_isSending;
92  }
93  // blocksize is default unless it's the last block, then it's the remainder
94  uint32_t blocksize = (m_numBlocks == 1) ? m_remainder : IR_DEFAULT_BLOCK_SIZE;
95  DEBUG_LOGF("IRSender Sending block #%u", m_numBlocks);
96  // pointer to the data at the correct location
97  const uint8_t *buf_ptr = m_bitStream.data() + m_writeCounter;
98  // iterate each byte in the block and write it
99  for (uint32_t i = 0; i < blocksize; ++i) {
100  // write the byte
101  sendByte(*buf_ptr);
102  // increment to next byte in buffer
103  buf_ptr++;
104  // record the written byte
105  m_writeCounter++;
106  }
107  // wrote one block
108  m_numBlocks--;
109  // still sending if we have more blocks
110  m_isSending = (m_numBlocks > 0);
111  // the curtime
112  m_lastSendTime = Time::getCurtime();
113  // return whether still sending
114  return m_isSending;
115 }

139 {
140  // Sends from left to right, MSB first

182  uint8_t pin = g_APinDescription[IR_SEND_PWM_PIN].ulPin; // 8
183  ETCChannel IR_TCC_Channel = TCC0_CH0;

185  //println();Serial.print("Port:"); Serial.print(port,DEC); Serial.print(" Pin:"); Serial.println(pin,DEC);
186  // Enable the port multiplexer for the PWM channel on pin
187  PORT->Group[port].PINCFG[pin].bit.PMUXEN = 1;
188 
189  // Connect the TCC timer to the port outputs - port pins are paired odd PMUXO and even PMUXEII
190  // F & E peripherals specify the timers: TCC0, TCC1 and TCC2
191  PORT->Group[port].PMUX[pin >> 1].reg |= IR_PER_EorF;
192 
193 // pinPeripheral (IR_SEND_PWM_PIN,PIO_TIMER_ALT);

196  GCLK_CLKCTRL_GEN_GCLK0 | // Select GCLK0
197  GCLK_CLKCTRL_ID_TCC0_TCC1; // Feed GCLK0 to TCC0 and TCC1
198  while (GCLK->STATUS.bit.SYNCBUSY); // Wait for synchronization
199 
200  // Normal (single slope) PWM operation: timers countinuously count up to PER
201  // register value and then is reset to 0
202  IR_TCCx = (Tcc*) GetTC(IR_TCC_Channel);
203  IR_TCCx->WAVE.reg |= TCC_WAVE_WAVEGEN_NPWM; // Setup single slope PWM on TCCx
204  while (IR_TCCx->SYNCBUSY.bit.WAVE); // Wait for synchronization
205

207  // this determines the frequency of the PWM operation.

210  while(IR_TCCx->SYNCBUSY.bit.PER);
211

213  // Set the duty cycle of the PWM on TCC0 to 33%
