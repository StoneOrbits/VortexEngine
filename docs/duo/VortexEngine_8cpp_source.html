169 void VortexEngine::runMainLogic()
170 {
171  // the current tick
172  uint32_t now = Time::getCurtime();
173 
174  // if the device is locked then that takes priority over all, while locked the
175  // device will only listen for clicks to wakeup momentarily then go back to sleep
176  if (Modes::locked()) {
177  // several fast clicks will unlock the device
178  if (g_pButton->onConsecutivePresses(DEVICE_LOCK_CLICKS - 1)) {
179  // turn off the lock flag and save it to disk
180  Modes::setLocked(false);
181 #ifdef VORTEX_EMBEDDED
182  // then enable the mosfet
183  enableMOSFET(true);
184 #endif
185  } else if (now > (CONSECUTIVE_WINDOW_TICKS * DEVICE_LOCK_CLICKS)) {
186  // go back to sleep if they don't unlock in time, don't save
187  enterSleep(false);
188  }
189  // OPTIONAL: render a dim led during unlock window waiting for clicks?
190  //Leds::setIndex(LED_1, RGB_RED4);
191  Leds::clearAll();
192  // don't do anything else while locked, just return
193  return;
194  }
195 
196  // the device is not locked, proceed with regular logic
197 
198  // if the button hasn't been released since turning on then there is custom logic
199  if (g_pButton->releaseCount() == 0) {
200  // if the button is held for 2 seconds from off, switch to on click mode on
201  // the last mode shown before sleep
202  if (!Modes::keychainModeEnabled() && now == ONE_CLICK_THRESHOLD_TICKS && g_pButton->isPressed()) {
203  // whether oneclick mode is now enabled
204  bool isEnabledNow = !Modes::oneClickModeEnabled();
205  // toggle one click mode
206  Modes::setOneClickMode(isEnabledNow);
207  // if we turned it on then switch to that mode
208  if (isEnabledNow) {
209  Modes::switchToStartupMode();
210  } else {
211  Modes::setCurMode(0);
212  }
213  // flash either low white or dim white2 to indicate
214  // whether one-click mode has been turned on or off
215  Leds::holdAll(isEnabledNow ? RGB_WHITE0 : RGB_WHITE5);
216  }
217  return;
218  }
219 
220 #ifdef VORTEX_EMBEDDED
221  // ESD PROTECTION!
222  // Sometimes the chip can be turned on via ESD triggering the wakeup pin
223  // if the engine makes it here in less than 2 ticks that means the device turned on
224  // via ESD and not via a normal click which cannot possibly be done in less than 1 tick
225  if (now < 2) {
226  // if that happens then just gracefully go back to sleep to prevent the chip
227  // from turning on randomly in a plastic bag
228  // do not save on ESD re-sleep
229  enterSleep(false);
230  return;
231  }
232 #endif
233 
234  // finally the user has released the button after initially turning it on,
235  // just run the regular main logic of the system
236 
237  // re-enter keychain mode if it was never disabled
238  if (Modes::keychainModeEnabled() && !Menus::checkInMenu()) {
239  // switch to the last mode we were on
240  Modes::switchToStartupMode();
241  // enter keychain mode menu
242  Menus::openMenu(MENU_GLOBAL_BRIGHTNESS, true);
243  }
244 
245  // first look for the force-sleep and instant on/off toggle
246  const uint32_t holdTime = g_pButton->holdDuration();
247  // force-sleep check takes precedence above all, but it does not run when keychain mode is enabled
248  if (m_forceSleepEnabled && holdTime >= FORCE_SLEEP_THRESHOLD_TICKS) {
249  // as long as they hold down past this threshold just turn off
250  if (g_pButton->isPressed()) {
251  Leds::clearAll();
252  return;
253  }
254  // but as soon as they actually release put the device to sleep
255  if (g_pButton->onRelease()) {
256  // do not save on force sleep
257  enterSleep(false);
258  }
259  return;
260  }
261 
262  // run the menus to see if they are open and need to do anything
263  if (Menus::run()) {
264  // if they return true that means the menus are open and rendering so just return
265  return;
266  }
267 
268  // if the user releases the button after the sleep threshold and
269  // we're still in menu state not open, then we can go to sleep
270  if (g_pButton->onRelease() && holdTime >= SLEEP_ENTER_THRESHOLD_TICKS) {
271  // enter sleep mode
272  enterSleep();
273  return;
274  }
275 
276  // if the button is just held beyond the sleep threshold then
277  if (g_pButton->isPressed() && holdTime >= SLEEP_ENTER_THRESHOLD_TICKS) {
278  // clear all the leds for a short moment
279  Leds::clearAll();
280  // then oncethe user holds past the sleep window threshold open up the menus
281  if (holdTime >= (SLEEP_ENTER_THRESHOLD_TICKS + SLEEP_WINDOW_THRESHOLD_TICKS)) {
282  // open the menu selection area
283  DEBUG_LOG("Entering ring fill...");
284  Menus::openMenuSelection();
285  }
286  // don't play the modes because the user is going into menus
287  return;
288  }
289 
290  // lastly check if we are locking the device, which can only happen if they click the
291  // button 5 times quickly when the device was off, so 4 times in the first x ticks because
292  // the first click was used to wake the device and isn't counted in the consecutive clicks
293  if (now < (CONSECUTIVE_WINDOW_TICKS * DEVICE_LOCK_CLICKS) && g_pButton->onConsecutivePresses(DEVICE_LOCK_CLICKS - 1)) {
294 #ifdef VORTEX_LIB
295  if (!Vortex::lockEnabled()) {
296  return;
297  }
298 #endif
299  // lock and just go to sleep, don't need to reset consecutive press counter here
300  Modes::setLocked(true);
301  enterSleep();
302  return;
303  }
304 
305  // toggle auto cycle mode with many clicks at main modes
306  if (g_pButton->onConsecutivePresses(AUTO_CYCLE_MODES_CLICKS)) {
307  m_autoCycle = !m_autoCycle;
308  Leds::holdAll(m_autoCycle ? RGB_GREEN : RGB_RED);
309  }
310 
311  // if auto cycle is enabled and the last switch was more than the delay ago
312  if (m_autoCycle && (Modes::lastSwitchTime() + AUTO_RANDOM_DELAY < now)) {
313  // then switch to the next mode automatically
314  Modes::nextModeSkipEmpty();
315  }
316 
317  // otherwise just play the modes
318  Modes::play();
319 }

359 void VortexEngine::enterSleep(bool save)
360 {
361  DEBUG_LOG("Sleeping");
362  if (save) {
363  // update the startup mode when going to sleep
364  Modes::setStartupMode(Modes::curModeIndex());
365  // save anything that hasn't been saved
366  Modes::saveStorage();
367  }
368  // clear all the leds
369  Leds::clearAll();
370  Leds::update();
371 #ifdef VORTEX_EMBEDDED
372  // init the output pins to prevent any floating pins
373  clearOutputPins();
374  // close the mosfet so that power cannot flow to the leds
375  enableMOSFET(false);
376  // delay for a bit to let the mosfet close and leds turn off
377  Time::delayMicroseconds(250);
378  // this is an ISR that runs in the timecontrol system to handle
379  // micros, it will wake the device up periodically
380  TCB0.INTCTRL = 0;
381  TCB0.CTRLA = 0;
382  // Enable wake on interrupt for the button
383  g_pButton->enableWake();
384  // Set sleep mode to POWER DOWN mode
385  set_sleep_mode(SLEEP_MODE_PWR_DOWN);
386  // enable the sleep boo lright before we enter sleep, this will allow
387  // the main loop to break and return
388  m_sleeping = true;
389  // enter sleep
390  sleep_mode();
391 #else
392  // enable the sleep bool
393  m_sleeping = true;
394 #endif
