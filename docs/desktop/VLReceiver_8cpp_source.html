163 // state machine that can be fed VL timings to parse them and interpret the intervals
164 void VLReceiver::handleVLTiming(uint32_t diff)
165 {
166  // if the diff is too long or too short then it's not useful
167  if ((diff > VL_HEADER_MARK_MAX && m_recvState < READING_DATA_MARK) || diff < VL_TIMING_MIN) {
168  DEBUG_LOGF("bad delay: %u, resetting...", diff);
169  resetVLState();
170  return;
171  }
172  switch (m_recvState) {
173  case WAITING_HEADER_MARK: // initial state
174  if (diff >= VL_HEADER_SPACE_MIN && diff <= VL_HEADER_MARK_MAX) {
175  m_recvState = WAITING_HEADER_SPACE;
176  } else {
177  DEBUG_LOGF("Bad header mark %u, resetting...", diff);
178  resetVLState();
179  }
180  break;
181  case WAITING_HEADER_SPACE:
182  if (diff >= VL_HEADER_SPACE_MIN && diff <= VL_HEADER_MARK_MAX) {
183  m_recvState = READING_DATA_MARK;
184  } else {
185  DEBUG_LOGF("Bad header space %u, resetting...", diff);
186  resetVLState();
187  }
188  break;
189  case READING_DATA_MARK:
190  // classify mark/space based on the timing and write into buffer
191  m_vlData.write1Bit((diff > (VL_TIMING * 2)) ? 1 : 0);
192  m_recvState = READING_DATA_SPACE;
193  break;
194  case READING_DATA_SPACE:
195  // the space could be just a regular space, or a gap in between blocks
196  m_recvState = READING_DATA_MARK;
197  break;
198  default: // ??
199  DEBUG_LOGF("Bad receive state: %u", m_recvState);
200  break;
201  }
202 }
