70 bool IRSender::send()
71 {
72  // must have at least one block to send
73  if (m_numBlocks < 1) {
74  m_isSending = false;
75  return m_isSending;
76  }
77  if (!m_isSending) {
78  beginSend();
79  }
80  if (m_lastSendTime > 0 && m_lastSendTime + IR_DEFAULT_BLOCK_SPACING > m_engine.time().getCurtime()) {
81  // don't send yet
82  return m_isSending;
83  }
84  // blocksize is default unless it's the last block, then it's the remainder
85  uint32_t blocksize = (m_numBlocks == 1) ? m_remainder : IR_DEFAULT_BLOCK_SIZE;
86  DEBUG_LOGF("IRSender Sending block #%u", m_numBlocks);
87  // pointer to the data at the correct location
88  const uint8_t *buf_ptr = m_bitStream.data() + m_writeCounter;
89  // iterate each byte in the block and write it
90  for (uint32_t i = 0; i < blocksize; ++i) {
91  // write the byte
92  sendByte(*buf_ptr);
93  // increment to next byte in buffer
94  buf_ptr++;
95  // record the written byte
96  m_writeCounter++;
97  }
98  // wrote one block
99  m_numBlocks--;
100  // still sending if we have more blocks
101  m_isSending = (m_numBlocks > 0);
102  // the curtime
103  m_lastSendTime = m_engine.time().getCurtime();
104  // return whether still sending
105  return m_isSending;
106 }

127 void IRSender::sendByte(uint8_t data)
128 {
129  // Sends from left to right, MSB first
130  for (int b = 0; b < 8; b++) {
131  // grab the bit of data at the indexspace
132  uint32_t bit = (data >> (7 - b)) & 1;
133  // send 3x timing size for 1s and 1x timing for 0
134  sendMark(IR_TIMING + (IR_TIMING * (2 * bit)));
135  // send 1x timing size for space
