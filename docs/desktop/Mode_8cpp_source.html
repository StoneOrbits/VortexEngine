162 bool Mode::loadFromBuffer(ByteStream &modeBuffer)
163 {
164  if (!modeBuffer.decompress()) {
165  // failed to decompress the mode
166  return false;
167  }
168  // reset the unserializer index before unserializing anything
169  modeBuffer.resetUnserializer();
170  uint8_t major = 0;
171  uint8_t minor = 0;
172  // unserialize the vortex version
173  if (!modeBuffer.unserialize8(&major)) {
174  return false;
175  }
176  if (!modeBuffer.unserialize8(&minor)) {
177  return false;
178  }
179  // check the version for incompatibility
180  if (!m_engine.checkVersion(major, minor)) {
181  // incompatible version
182  ERROR_LOGF("Incompatible savefile version: %u.%u", major, minor);
183  return false;
184  }
185  // now just unserialize the list of patterns
186  if (!unserialize(modeBuffer)) {
187  return false;
188  }
189  // then initialize the mode so that it is ready to play
190  init();
191  return true;
192 }

194 bool Mode::serialize(ByteStream &buffer, uint8_t numLeds) const
195 {
196  if (!numLeds) {
197  numLeds = MODE_LED_COUNT;
198  }
199  // serialize the number of leds
200  if (!buffer.serialize8(numLeds)) {
201  return false;
202  }
203  // empty mode?
204  if (!numLeds) {
205  return true;
206  }
207  // serialize the flags
208  ModeFlags flags = getFlags();
209  if (!buffer.serialize8(flags)) {
210  return false;
211  }
212 #if VORTEX_SLIM == 0
213  // serialiaze the multi led?
214  if ((flags & MODE_FLAG_MULTI_LED) && m_multiPat) {
215  // serialize the multi led
216  if (!m_multiPat->serialize(buffer)) {
217  return false;
218  }
219  }
220 #endif
221  // if no single leds then just stop here
222  if (!(flags & MODE_FLAG_SINGLE_LED)) {
223  return true;
224  }
225  // if there are any sparse singles (spaces) then we need to
226  // serialize an led map of which singles are set
227  if (flags & MODE_FLAG_SPARSE_SINGLES) {
228  if (!buffer.serialize32((uint32_t)getSingleLedMap())) {
229  return false;
230  }
231  }
232  // then iterate each single led and serialize it
233  for (LedPos pos = LED_FIRST; pos < numLeds; ++pos) {
234  const Pattern *entry = m_singlePats[pos];
235  if (!entry) {
236  continue;
237  }
238  // just serialize the pattern then colorset
239  if (!entry->serialize(buffer)) {
240  return false;
241  }
242  // if they are all same single then only serialize one
243  if (flags & MODE_FLAG_ALL_SAME_SINGLE) {
244  break;
245  }
246  }
247  return true;
248 }

250 // this is a hairy function, but a bit of a necessary complexity
251 bool Mode::unserialize(ByteStream &buffer)
252 {
253  clearPattern(MODE_LED_ALL);
254  uint8_t ledCount = MODE_LED_COUNT;
255  // unserialize the number of leds
256  if (!buffer.unserialize8(&ledCount)) {
257  return false;
258  }
259  // adjust the internal led count of the mode if the incoming stream is different
260  // generally no embedded device would ever used an unfixed led count, this is purely
261  // for vortexlib to allow things like the editor to load modes of any size leds
262  if (ledCount != MODE_LED_COUNT) {
263  //setLedCount(ledCount);
264  }
265  if (!ledCount) {
266  // empty mode?
267  return true;
268  }
269  // unserialize the flags value
270  ModeFlags flags = 0;
271  if (!buffer.unserialize8(&flags)) {
272  return false;
273  }
274  Pattern *firstPat = nullptr;
275  // if there is a multi led pattern then unserialize it
276  if (flags & MODE_FLAG_MULTI_LED) {
277 #if VORTEX_SLIM == 1
278  // unserialize the multi pattern
279  Pattern *multiPat = m_engine.patternBuilder().unserialize(buffer);
280  // if there are no single leds then discard the firstpat
281  if ((flags & MODE_FLAG_SINGLE_LED) != 0 && multiPat) {
282  // discard the multi pattern
283  delete multiPat;
284  } else {
285  // otherwise turn on the all same single flag to use the multi as a single
286  flags = MODE_FLAG_SINGLE_LED | MODE_FLAG_ALL_SAME_SINGLE;
287  firstPat = multiPat;
288  }
289 #else
290  // otherwise in normal build actually unserialize it
291  m_multiPat = m_engine.patternBuilder().unserialize(buffer);
292  if (!m_multiPat) {
293  return false;
294  }
295  m_multiPat->init();
296 #endif
297  }
298  // if there is no single led patterns just stop here
299  if (!(flags & MODE_FLAG_SINGLE_LED)) {
300  return true;
301  }
302  // is there an led map to unserialize? if not default to all
303  LedMap map = (1 << ledCount) - 1;
304  if (flags & MODE_FLAG_SPARSE_SINGLES) {
305  if (!buffer.unserialize32((uint32_t *)&map)) {
306  return false;
307  }
308  }
309  // unserialize all singleled patterns into their positions
310  MAP_FOREACH_LED(map) {
311  if (pos >= MODE_LED_COUNT) {
312  // in case the map encodes led positions this device doesn't support
313  break;
314  }
315  if (!firstPat) {
316  // save the first pattern so that it can be duped if this is 'all same'
317  m_singlePats[pos] = firstPat = m_engine.patternBuilder().unserialize(buffer);
318  } else if (flags & MODE_FLAG_ALL_SAME_SINGLE) {
319  // if all same then just dupe first
320  m_singlePats[pos] = m_engine.patternBuilder().dupe(firstPat);
321  } else {
322  // otherwise unserialize the pattern like normal
323  m_singlePats[pos] = m_engine.patternBuilder().unserialize(buffer);
324  }
325  if (!m_singlePats[pos]) {
326  clearPattern(MODE_LED_ALL);
327  return false;
328  }
329  if (!m_singlePats[pos]) {
330  clearPattern(MODE_LED_ALL);
331  return false;
332  }
333  m_singlePats[pos]->bind(pos);
334  }
335  // there is a few different possibilities here:
336  // 1. The provided ledCount is less than our current MODE_LED_COUNT
337  // -> if this happens we need to repeat the first ledCount leds
338  // into the remaining MODE_LED_COUNT - ledCount
339  // 2. The provided ledCount is more than our current MODE_LED_COUNT
340  // -> if this happens then we can just chop the leds and continue
341  // 3. The provided ledCount is the same as our current MODE_LED_COUNT
342  // -> if this happens all is good we can continue
343  if (ledCount >= MODE_LED_COUNT) {
344  // in this case we either chopped some off or have the exact amount
345  return true;
346  }
347  // in this case we have to repeat them so we loop MODE_LED_COUNT - ledCount
348  // times and walk the first ledCount that are already set and copy them
349  LedPos src = LED_FIRST;
350  // start from ledCount (the first index we didn't load) and loop till
351  // MODE_LED_COUNT and dupe the pattern in the src position, but wrap the src
352  // around at ledCount so that we repeat the first ledCount over again
353  for (LedPos pos = (LedPos)ledCount; pos < MODE_LED_COUNT; ++pos) {
354  m_singlePats[pos] = m_engine.patternBuilder().dupe(m_singlePats[src]);
355  m_singlePats[pos]->bind(pos);
356  src = (LedPos)((src + 1) % ledCount);
357  }
358  return true;
359 }

361 bool Mode::equals(const Mode *other) const
362 {
363  if (!other) {
364  return false;
365  }
366  // compare the led count
367  if (other->getLedCount() != getLedCount()) {
368  return false;
369  }
370 #if VORTEX_SLIM == 0
371  // compare the multi pattern, either:
372  // have a local pat and it's not equal to other (other could be null)
373  // or have no local pat and other has one
374  if ((m_multiPat && !m_multiPat->equals(other->m_multiPat)) ||
375  (!m_multiPat && other->m_multiPat)) {
376  return false;
377  }
378 #endif
379  // compare all the singles
380  for (LedPos pos = LED_FIRST; pos < MODE_LED_COUNT; ++pos) {
381  // if entry is valid, do a comparison
382  if (m_singlePats[pos]) {
383  // checks if other is not null and equal
384  if (!m_singlePats[pos]->equals(other->m_singlePats[pos])) {
385  return false;
386  }
387  continue;
388  }
389  // current is null, check if other is valid
390  if (other->m_singlePats[pos]) {
391  return false;
392  }
393  // both are null
394  }
395  return true;

428 {
429  // equivalent to case MODE_LED_ALL
430  if (pos == MODE_LED_ALL) {
431  // makes no sense
432  return nullptr;
433  }
434 
435  // equivalent to case MODE_LED_ANY and MODE_LED_MULTI
436  else if (pos == MODE_LED_ANY || pos == MODE_LED_MULTI) {

455  }
456  }

461  return nullptr;

465  return m_singlePats[p];
466  }
467  }
468  }
