411 // stolen from FastLED raw hsv to rgb conversion, idk what is special
412 // about this one but I think the hue stops at 190 maybe?
413 RGBColor hsv_to_rgb_raw_C(const HSVColor &rhs)
414 {
415  RGBColor col;
416  // Convert hue, saturation and brightness ( HSV/HSB ) to RGB
417  // "Dimming" is used on saturation and brightness to make
418  // the output more visually linear.
419 
420  // Apply dimming curves
421  uint8_t value = rhs.val;
422  uint8_t saturation = rhs.sat;
423 
424  // The brightness floor is minimum number that all of
425  // R, G, and B will be set to.
426  uint8_t invsat = 255 - saturation;
427  uint8_t brightness_floor = (value * invsat) / 256;
428 
429  // The color amplitude is the maximum amount of R, G, and B
430  // that will be added on top of the brightness_floor to
431  // create the specific hue desired.
432  uint8_t color_amplitude = value - brightness_floor;
433 
434  // Figure out which section of the hue wheel we're in,
435  // and how far offset we are withing that section
436  uint8_t section = rhs.hue / 0x40; // 0..2
437  uint8_t offset = rhs.hue % 0x40; // 0..63
438 
439  uint8_t rampup = offset; // 0..63
440  uint8_t rampdown = (0x40 - 1) - offset; // 63..0
441 
442  // We now scale rampup and rampdown to a 0-255 range -- at least
443  // in theory, but here's where architecture-specific decsions
444  // come in to play:
445  // To scale them up to 0-255, we'd want to multiply by 4.
446  // But in the very next step, we multiply the ramps by other
447  // values and then divide the resulting product by 256.
448  // So which is faster?
449  // ((ramp * 4) * othervalue) / 256
450  // or
451  // ((ramp ) * othervalue) / 64
452  // It depends on your processor architecture.
453  // On 8-bit AVR, the "/ 256" is just a one-cycle register move,
454  // but the "/ 64" might be a multicycle shift process. So on AVR
455  // it's faster do multiply the ramp values by four, and then
456  // divide by 256.
457  // On ARM, the "/ 256" and "/ 64" are one cycle each, so it's
458  // faster to NOT multiply the ramp values by four, and just to
459  // divide the resulting product by 64 (instead of 256).
460  // Moral of the story: trust your profiler, not your insticts.
461 
462  // Since there's an AVR assembly version elsewhere, we'll
463  // assume what we're on an architecture where any number of
464  // bit shifts has roughly the same cost, and we'll remove the
465  // redundant math at the source level:
466 
467  // // scale up to 255 range
468  // //rampup *= 4; // 0..252
469  // //rampdown *= 4; // 0..252
470 
471  // compute color-amplitude-scaled-down versions of rampup and rampdown
472  uint8_t rampup_amp_adj = (rampup * color_amplitude) / (256 / 4);
473  uint8_t rampdown_amp_adj = (rampdown * color_amplitude) / (256 / 4);
474 
475  // add brightness_floor offset to everything
476  uint8_t rampup_adj_with_floor = rampup_amp_adj + brightness_floor;
477  uint8_t rampdown_adj_with_floor = rampdown_amp_adj + brightness_floor;
478 
479  if (section) {
480  if (section == 1) {
481  // section 1: 0x40..0x7F
482  col.red = brightness_floor;
483  col.green = rampdown_adj_with_floor;
484  col.blue = rampup_adj_with_floor;
485  } else {
486  // section 2; 0x80..0xBF
487  col.red = rampup_adj_with_floor;
488  col.green = brightness_floor;
489  col.blue = rampdown_adj_with_floor;
490  }
491  } else {
492  // section 0: 0x00..0x3F
493  col.red = rampdown_adj_with_floor;
494  col.green = rampup_adj_with_floor;
495  col.blue = brightness_floor;
496  }
497  return col;
498 }

544 static uint8_t sqrt16(uint16_t x)
545 {
546  if (x <= 1) {
547  return (uint8_t)x;
548  }
549  uint8_t low = 1; // lower bound
550  uint8_t mid = 0;
551  // initial estimate for upper bound
552  uint8_t hi = (x > 7904) ? 255 : ((x >> 5) + 8);
553  do {
554  mid = (low + hi) >> 1;
555  if ((uint16_t)(mid * mid) > x) {

560  }
561  low = mid + 1;
562  }

588 // Convert rgb to hsv with slower approximation that matches
589 // the fastled hsv to rgb 'rainbow'
590 HSVColor rgb_to_hsv_approx(const RGBColor &rhs)
591 {
592  uint8_t r = rhs.red;
593  uint8_t g = rhs.green;
594  uint8_t b = rhs.blue;
595  uint8_t h, s, v;
596  // find desaturation
597  uint8_t desat = 255;
598  if (r < desat) desat = r;
599  if (g < desat) desat = g;
600  if (b < desat) desat = b;
601  // remove saturation from all channels
602  r -= desat;
603  g -= desat;
604  b -= desat;
605  // saturation is opposite of desaturation
606  s = 255 - desat;
607  if (s != 255) {
608  // undo 'dimming' of saturation
609  s = 255 - sqrt16((255 - s) * 256);
610  }
611  // at least one channel is now zero
612  // if all three channels are zero, we had a
613  // shade of gray.
614  if ((r + g + b) == 0) {
615  // we pick hue zero for no special reason
616  return HSVColor(0, 0, 255 - s);
617  }
618  // scale all channels up to compensate for desaturation
619  if (s < 255) {
620  if (s == 0) s = 1;
621  uint32_t scaleup = 65535 / (s);
622  r = ((uint32_t)(r)*scaleup) / 256;
623  g = ((uint32_t)(g)*scaleup) / 256;
624  b = ((uint32_t)(b)*scaleup) / 256;
625  }
626  uint16_t total = r + g + b;
627  // scale all channels up to compensate for low values
628  if (total < 255) {
629  if (total == 0) total = 1;
630  uint32_t scaleup = 65535 / (total);
631  r = ((uint32_t)(r)*scaleup) / 256;
632  g = ((uint32_t)(g)*scaleup) / 256;
633  b = ((uint32_t)(b)*scaleup) / 256;
634  }
635  if (total > 255) {
636  v = 255;
637  } else {
638  if (((uint32_t)desat + (uint32_t)total) <= 255) {
639  v = desat + total;
640  } else {
641  v = 255;
642  }
643  // undo 'dimming' of brightness
644  if (v != 255) v = sqrt16(v * 256);
645  }

647  // (ties don't matter)
648  uint8_t highest = r;
649  if (g > highest) highest = g;
650  if (b > highest) highest = b;
651  if (highest == r) {
