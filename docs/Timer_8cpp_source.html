109 AlarmID Timer::alarm()
110 {
111  if (!m_numAlarms || !m_alarms || m_curAlarm == ALARM_NONE) {
112  return ALARM_NONE;
113  }
114  uint32_t now = Time::getCurtime();
115  // time since start (forward or backwards)
116  int32_t timeDiff = (int32_t)(int64_t)(now - getStartTime());
117  if (timeDiff < 0) {
118  return ALARM_NONE;
119  }
120  // if no time passed it's first alarm that is starting
121  if (timeDiff == 0) {
122  return 0;
123  }
124  uint32_t alarmTime = m_alarms[m_curAlarm];
125  // if the current alarm duration is not a multiple of the current tick
126  if (alarmTime && (timeDiff % alarmTime) != 0) {
127  // then the alarm was not hit
128  return ALARM_NONE;
129  }
130  // update the start time of the timer
131  setStartTime(now);
132  // increment current alarm then return that id
133  m_curAlarm = (AlarmID)((m_curAlarm + 1) % m_numAlarms);
134  return m_curAlarm;
